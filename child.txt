#include <stdio.h>
#include <unistd.h>
int main() {
    pid_t child_pid;
    child_pid = fork();
    if (child_pid == -1) {
        perror("fork");
        return 1;
    } else if (child_pid == 0) {
        printf("Child process: My PID is %d\n", getpid());
        printf("Child process: My parent's PID is %d\n", getppid());
    } else {
        printf("Parent process: My PID is %d\n", getpid());
        printf("Parent process: My parent's PID is %d\n", getppid());
    }
    return 0;
}
